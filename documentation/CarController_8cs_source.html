135 private void ApplyDrive(float accel, float footbrake)
136 {
137 float thrustTorque;
138 switch (m_CarDriveType)
139 {
140 case CarDriveType.FourWheelDrive:
141 thrustTorque = accel * (m_CurrentTorque / 4f);

144 m_WheelColliders[i].motorTorque = thrustTorque;
145 }
146 break;
147
148 case CarDriveType.FrontWheelDrive:
149 thrustTorque = accel * (m_CurrentTorque / 2f);
150 m_WheelColliders[0].motorTorque = m_WheelColliders[1].motorTorque = thrustTorque;
151 break;
152
153 case CarDriveType.RearWheelDrive:
154 thrustTorque = accel * (m_CurrentTorque / 2f);
155 m_WheelColliders[2].motorTorque = m_WheelColliders[3].motorTorque = thrustTorque;
156 break;
157 }
158
159 for (int i = 0; i < 4; i++)
160 {
161 if (CurrentSpeed > 5 && Vector3.Angle(transform.forward, m_Rigidbody.velocity) < 50f)
162 {
163 m_WheelColliders[i].brakeTorque = m_BrakeTorque * footbrake;
164 }
165 else if (footbrake > 0)

173 private void SteerHelper()
174 {
175 for (int i = 0; i < 4; i++)
176 {
177 WheelHit wheelhit;
178 m_WheelColliders[i].GetGroundHit(out wheelhit);
179 if (wheelhit.normal == Vector3.zero)
180 return; // wheels arent on the ground so dont realign the rigidbody velocity
181 }
182
183 // this if is needed to avoid gimbal lock problems that will make the car suddenly shift direction
184 if (Mathf.Abs(m_OldRotation - transform.eulerAngles.y) < 10f)
185 {
186 var turnadjust = (transform.eulerAngles.y - m_OldRotation) * m_SteerHelper;
187 Quaternion velRotation = Quaternion.AngleAxis(turnadjust, Vector3.up);
188 m_Rigidbody.velocity = velRotation * m_Rigidbody.velocity;
189 }
190 m_OldRotation = transform.eulerAngles.y;
191 }
