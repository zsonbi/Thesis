219 private void DespawnFarAwayChunks(int row, int col)
220 {
221 for (int i = -(chunkLoadDistance + 1); i <= chunkLoadDistance + 1; i++)
222 {
223 for (int j = -(chunkLoadDistance + 1); j <= chunkLoadDistance + 1; j++)
224 {
225 //If the chunk is close enough skip it
226 if (Mathf.Abs(i) <= chunkLoadDistance && Mathf.Abs(j) <= chunkLoadDistance)
227 {
228 continue;
229 }
230 //If the chunk is far away unload it
231 if (i + row >= 0 && j + col >= 0 && i + row < GameConfig.CHUNK_COUNT && j + col < GameConfig.CHUNK_COUNT)
232 {
233 if (world is not null)
234 world.HideChunk(j + col, i + row);
235 }
236 }
237 }
238 }

245 private async Task SpawnNearbyChunks(int row, int col)

260 for (int y = 0 - currSize; y <= 0 + currSize; y += currSize * 2)
261 {
262 for (int x = 0; x <= currSize - 1; x++)
263 {
264 await ValidateAndLoadChunk(row + y, col + x);
265 await ValidateAndLoadChunk(row + y, col - x);
266 }
267 }
268

270 await ValidateAndLoadChunk(row - currSize, col - currSize);
271 await ValidateAndLoadChunk(row + currSize, col - currSize);
