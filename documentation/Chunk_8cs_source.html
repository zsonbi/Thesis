389 private ChunkCellContainer DetermineRoadType(int col, int row)
390 {
391 int roadCount = 0;
392
393 int firstIndex = -1;
394 int lastIndex = -1;
395 int secondIndex = -1;
396
397 if (col - 1 >= 0 && roadGenerator.RoadMatrix[row, col - 1])
398 {
399 roadCount++;
400 firstIndex = 0;
401 lastIndex = 0;
402 }
403
404 if (row + 1 < GameConfig.CHUNK_SIZE && roadGenerator.RoadMatrix[row + 1, col])
405 {
406 roadCount++;
407
408 if (firstIndex == -1)
409 {
410 firstIndex = 1;
411 }
412 else
413 {
414 secondIndex = 1;
415 }
416 lastIndex = 1;
417 }
418
419 if (col + 1 < GameConfig.CHUNK_SIZE && roadGenerator.RoadMatrix[row, col + 1])
420 {
421 roadCount++;
422
423 if (firstIndex == -1)
424 {
425 firstIndex = 2;
426 }
427 else if (secondIndex == -1)
428 {
429 secondIndex = 2;
430 }
431 lastIndex = 2;
432 }
433
434 if (row - 1 >= 0 && roadGenerator.RoadMatrix[row - 1, col])
435 {
436 roadCount++;
437
438 if (firstIndex == -1)
439 {
440 firstIndex = 3;
441 }
442 else if (secondIndex == -1)
443 {
444 secondIndex = 3;
445 }
446 lastIndex = 3;
447 }
448
449 if (roadCount == 4)
450 {
451 return new ChunkCellContainer(ChunkCellType.CrossRoads, Vector3.zero);
452 }
453 else if (roadCount == 3)
454 {
455 if (firstIndex == 0 && lastIndex == 3)
456 {
457 if (secondIndex == 1)
458 {
459 return new ChunkCellContainer(ChunkCellType.ThreeWay, new Vector3(0, (1) * 90, 0));
460 }
461 else
462 {
463 return new ChunkCellContainer(ChunkCellType.ThreeWay, Vector3.zero);
464 }
465 }
466
467 return new ChunkCellContainer(ChunkCellType.ThreeWay, new Vector3(0, (secondIndex + 1) * 90, 0));
468 }
469 else if (roadCount == 2 && (lastIndex - firstIndex) % 2 == 1)
470 {
471 if (secondIndex == 3 && firstIndex == 0)
472 {
473 return new ChunkCellContainer(ChunkCellType.CurvedRoad, new Vector3(0, (-1) * 90, 0));
474 }
475
476 return new ChunkCellContainer(ChunkCellType.CurvedRoad, new Vector3(0, (lastIndex - 1) * 90, 0));
477 }
478 else
479 {
480 return new ChunkCellContainer(ChunkCellType.StraightRoad, new Vector3(0, (lastIndex % 2 + 1) * 90, 0));
481 }

535 private void GenerateCellMatrix()
536 {
537 foreach (var road in roads)
538 {
539 buildingCells[road.z, road.x] = new BuildingCell(BuildingDirection.None, false);
540
541 for (var i = road.z - 1; i <= road.z + 1; i += 2)
542 {
543 if (i < 0 || i >= GameConfig.CHUNK_SIZE || buildingCells[i, road.x] != null)
544 {
545 continue;
546 }
547 if (roadGenerator.RoadMatrix[i, road.x])
548 {
549 buildingCells[i, road.x] = new BuildingCell(BuildingDirection.None, false);
550 }
551 else
552 {
553 BuildingDirection roadDir = BuildingDirection.None;
554 if (i > road.z)
555 {
556 roadDir = BuildingDirection.Down;
557 }
558 else if (i < road.z)
559 {
560 roadDir = BuildingDirection.Up;
561 }
562
563 buildingCells[i, road.x] = new BuildingCell(roadDir, true);
564 }
565 }
566
567 for (int j = road.x - 1; j <= road.x + 1; j += 2)
568 {
569 if (j < 0 || j >= GameConfig.CHUNK_SIZE || buildingCells[road.z, j] != null)
570 {
571 continue;
572 }
573 if (roadGenerator.RoadMatrix[road.z, j])
574 {
575 buildingCells[road.z, j] = new BuildingCell(BuildingDirection.None, false);
576 }
577 else
578 {
579 BuildingDirection roadDir = BuildingDirection.None;
580
581 if (j > road.x)
582 {
583 roadDir = BuildingDirection.Left;
584 }
585 else if (j < road.x)
586 {
587 roadDir = BuildingDirection.Right;
588 }
589
590 buildingCells[road.z, j] = new BuildingCell(roadDir, true);
591 }
592 }
593 }
594
595 for (int i = 0; i < GameConfig.CHUNK_SIZE; i++)
596 {
597 for (int j = 0; j < GameConfig.CHUNK_SIZE; j++)
598 {
599 if (buildingCells[i, j] is null)
600 {
601 buildingCells[i, j] = new BuildingCell(BuildingDirection.None, true);

673 private void LockAndTryToPlaceBuilding(int index, int row, int col)
674 {
675 if (buildingCells[row, col].GotRoadNext && buildingCells[row, col].Buildable)
676 {
677 Building building = this.buildings[index];
678 List<BuildingCell> neighbourBuildingCells = new List<BuildingCell>();
679 float zPosCorrection = 0f;
680 float xPosCorrection = 0f;
681 if (((int)buildingCells[row, col].RoadDirection) % 2 == 1)
682 {
683 for (int i = building.RowCount / -2; i < Mathf.CeilToInt(building.RowCount / 2f); i++)
684 {
685 if (!buildingCells[row, col + i].Buildable)
686 {
687 return;
688 }
689 neighbourBuildingCells.Add(buildingCells[row, col + i]);
690 }
691 if (building.RowCount % 2 == 0)
692 {
693 xPosCorrection = GameConfig.CHUNK_SIZE / -2;
694 }
695 if (building.ColumnCount % 2 == 0)
696 {
697 zPosCorrection = GameConfig.CHUNK_SIZE / 2;
698 }
699 }
700 else
701 {
702 for (int i = building.RowCount / -2; i < Mathf.CeilToInt(building.RowCount / 2f); i++)
703 {
704 if (!buildingCells[row + i, col].Buildable)
705 {
706 return;
707 }
708 neighbourBuildingCells.Add(buildingCells[row + i, col]);
709 }
710
711 if (building.RowCount % 2 == 0)
712 {
713 zPosCorrection = GameConfig.CHUNK_SIZE / -2;
714 }
715 if (building.ColumnCount % 2 == 0)
716 {
717 xPosCorrection = GameConfig.CHUNK_SIZE / 2;
718 }
719 }
720
721 if (Random.Range(0, 2) == 0)
722 {
723 float rotation = ((int)buildingCells[row, col].RoadDirection) % 4 * 90f;
724 var absolutePosition = GetAbsolutePosition();
725 Vector3 postition = new Vector3(absolutePosition.Item1 + col * GameConfig.CHUNK_SCALE * GameConfig.CHUNK_CELL + xPosCorrection, 0, absolutePosition.Item2 + row * GameConfig.CHUNK_SCALE * GameConfig.CHUNK_CELL + zPosCorrection);
726
727 Addressables.InstantiateAsync(building.AddressableKey, postition, Quaternion.Euler(0, rotation, 0), this.gameObject.transform);
728
729 foreach (var item in neighbourBuildingCells)
730 {
731 item.Occupy();
732 }
733 }
734 }
