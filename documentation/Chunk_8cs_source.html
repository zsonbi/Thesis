188 private void CombineMeshes(ChunkCellType cellType)
189 {
190 MeshRenderer[] filters = objectsToCombine[cellType][0].GetComponentsInChildren<MeshRenderer>();
191 int itr = 0;
192 foreach (MeshRenderer filter in filters)
193 {
194 //Creates the parent which will have the combined mesh
195 GameObject parent = new GameObject(objectsToCombine[cellType][0].name.Replace("(Clone)", "") + "Mesh", typeof(MeshFilter), typeof(MeshRenderer), typeof(MeshCollider));
196 MeshFilter meshFilter = parent.GetComponent<MeshFilter>();
197 //Assign the material of the first gameobject in the list to the parent
198 parent.GetComponent<MeshRenderer>().material = filter.material;
199 parent.transform.parent = this.transform;
200
201 meshFilter.mesh = new Mesh();
202 //Makes so that really big meshes are supported also
203 meshFilter.mesh.indexFormat = UnityEngine.Rendering.IndexFormat.UInt32;
204 CombineInstance[] combine = new CombineInstance[objectsToCombine[cellType].Count];
205 for (int i = 0; i < objectsToCombine[cellType].Count; i++)
206 {
207 MeshFilter[] objectMeshFilters = objectsToCombine[cellType][i].GetComponentsInChildren<MeshFilter>();
208 combine[i].mesh = objectMeshFilters[itr].sharedMesh;
209 combine[i].transform = objectMeshFilters[itr].transform.localToWorldMatrix;
210 }
211 meshFilter.mesh.CombineMeshes(combine, true, true);
212 meshFilter.gameObject.SetActive(true);
213
214 MeshCollider collider = parent.GetComponent<MeshCollider>();
215 collider.sharedMesh = meshFilter.sharedMesh;
216 parent.isStatic = true;
217
218 itr++;
219 }
220
221 //Empties the list and deletes the no longer used gameobjects
222 do
223 {
224 Destroy(objectsToCombine[cellType][0]);
225 objectsToCombine[cellType].RemoveAt(0);
226 } while (objectsToCombine[cellType].Count > 0);

460 private void GenerateCellMatrix()
461 {
462 foreach (var road in roads)
463 {
464 buildingCells[road.z, road.x] = new BuildingCell(BuildingDirection.None, false);
465
466 for (var i = road.z - 1; i <= road.z + 1; i += 2)
467 {
468 if (i < 0 || i >= GameConfig.CHUNK_SIZE || buildingCells[i, road.x] != null)
469 {
470 continue;
471 }
472 if (roadGenerator.RoadMatrix[i, road.x])
473 {
474 buildingCells[i, road.x] = new BuildingCell(BuildingDirection.None, false);
475 }
476 else
477 {
478 BuildingDirection roadDir = BuildingDirection.None;
479 if (i > road.z)
480 {
481 roadDir = BuildingDirection.Down;
482 }
483 else if (i < road.z)
484 {
485 roadDir = BuildingDirection.Up;
486 }
487
488 buildingCells[i, road.x] = new BuildingCell(roadDir, true);
489 }
490 }
491
492 for (int j = road.x - 1; j <= road.x + 1; j += 2)
493 {
494 if (j < 0 || j >= GameConfig.CHUNK_SIZE || buildingCells[road.z, j] != null)
495 {
496 continue;
497 }
498 if (roadGenerator.RoadMatrix[road.z, j])
499 {
500 buildingCells[road.z, j] = new BuildingCell(BuildingDirection.None, false);
501 }
502 else
503 {
504 BuildingDirection roadDir = BuildingDirection.None;
505
506 if (j > road.x)
507 {
508 roadDir = BuildingDirection.Left;
509 }
510 else if (j < road.x)
511 {
512 roadDir = BuildingDirection.Right;
513 }
514
515 buildingCells[road.z, j] = new BuildingCell(roadDir, true);
516 }
517 }
518 }
519
520 for (int i = 0; i < GameConfig.CHUNK_SIZE; i++)
521 {
522 for (int j = 0; j < GameConfig.CHUNK_SIZE; j++)
523 {
524 if (buildingCells[i, j] is null)
525 {
526 buildingCells[i, j] = new BuildingCell(BuildingDirection.None, true);
527 }
528 }

543 private void LockAndTryToPlaceBuilding(int index, int row, int col)
544 {
545 if (buildingCells[row, col].GotRoadNext && buildingCells[row, col].Buildable)
546 {
547 Building building = this.buildings[index];
548 List<BuildingCell> neighbourBuildingCells = new List<BuildingCell>();
549 float zPosCorrection = 0f;
550 float xPosCorrection = 0f;
551 if (((int)buildingCells[row, col].RoadDirection) % 2 == 1)
552 {
553 for (int i = building.RowCount / -2; i < Mathf.CeilToInt(building.RowCount / 2f); i++)
554 {
555 if (!buildingCells[row, col + i].Buildable)
556 {
557 return;
558 }
559 neighbourBuildingCells.Add(buildingCells[row, col + i]);
560 }
561 if (building.RowCount % 2 == 0)
562 {
563 xPosCorrection = GameConfig.CHUNK_SIZE / -2;
564 }
565 if (building.ColumnCount % 2 == 0)
566 {
567 zPosCorrection = GameConfig.CHUNK_SIZE / 2;
568 }
569 }
570 else
571 {
572 for (int i = building.RowCount / -2; i < Mathf.CeilToInt(building.RowCount / 2f); i++)
573 {
574 if (!buildingCells[row + i, col].Buildable)
575 {
576 return;
577 }
578 neighbourBuildingCells.Add(buildingCells[row + i, col]);
579 }
580
581 if (building.RowCount % 2 == 0)
582 {
583 zPosCorrection = GameConfig.CHUNK_SIZE / -2;
584 }
585 if (building.ColumnCount % 2 == 0)
586 {
587 xPosCorrection = GameConfig.CHUNK_SIZE / 2;
588 }
589 }
590
591 if (Random.Range(0, 3) == 0)
592 {
593 float rotation = ((int)buildingCells[row, col].RoadDirection) % 4 * 90f;
594 var absolutePosition = GetAbsolutePosition();
595 Vector3 postition = new Vector3(absolutePosition.Item1 + col * GameConfig.CHUNK_SCALE * GameConfig.CHUNK_CELL + xPosCorrection, 0, absolutePosition.Item2 + row * GameConfig.CHUNK_SCALE * GameConfig.CHUNK_CELL + zPosCorrection);
596
597 Addressables.InstantiateAsync(building.AddressableKey, postition, Quaternion.Euler(0, rotation, 0), this.gameObject.transform);
598
599 foreach (var item in neighbourBuildingCells)
600 {
601 item.Occupy();
602 }
603 }
604 }
