185 private void CombineMeshes(ChunkCellType cellType)
186 {
187 MeshRenderer[] filters = objectsToCombine[cellType][0].GetComponentsInChildren<MeshRenderer>();
188 int itr = 0;
189 foreach (MeshRenderer filter in filters)
190 {
191 //Creates the parent which will have the combined mesh
192 GameObject parent = new GameObject(objectsToCombine[cellType][0].name.Replace("(Clone)", "") + "Mesh", typeof(MeshFilter), typeof(MeshRenderer), typeof(MeshCollider));
193 MeshFilter meshFilter = parent.GetComponent<MeshFilter>();
194 //Assign the material of the first gameobject in the list to the parent
195 parent.GetComponent<MeshRenderer>().material = filter.material;
196 parent.transform.parent = this.transform;
197
198 meshFilter.mesh = new Mesh();
199 //Makes so that really big meshes are supported also
200 meshFilter.mesh.indexFormat = UnityEngine.Rendering.IndexFormat.UInt32;
201 CombineInstance[] combine = new CombineInstance[objectsToCombine[cellType].Count];
202 for (int i = 0; i < objectsToCombine[cellType].Count; i++)
203 {
204 MeshFilter[] objectMeshFilters = objectsToCombine[cellType][i].GetComponentsInChildren<MeshFilter>();
205 combine[i].mesh = objectMeshFilters[itr].sharedMesh;
206 combine[i].transform = objectMeshFilters[itr].transform.localToWorldMatrix;
207 }
208 meshFilter.mesh.CombineMeshes(combine, true, true);
209 meshFilter.gameObject.SetActive(true);
210
211 MeshCollider collider = parent.GetComponent<MeshCollider>();
212 collider.sharedMesh = meshFilter.sharedMesh;
213 parent.isStatic = true;
214
215 itr++;
216 }
217
218 //Empties the list and deletes the no longer used gameobjects
219 do
220 {
221 Destroy(objectsToCombine[cellType][0]);
222 objectsToCombine[cellType].RemoveAt(0);
223 } while (objectsToCombine[cellType].Count > 0);

451 private void GenerateCellMatrix()
452 {
453 foreach (var road in roads)
454 {
455 buildingCells[road.z, road.x] = new BuildingCell(BuildingDirection.None, false);
456
457 for (var i = road.z - 1; i <= road.z + 1; i += 2)
458 {
459 if (i < 0 || i >= GameConfig.CHUNK_SIZE || buildingCells[i, road.x] != null)
460 {
461 continue;
462 }
463 if (roadGenerator.RoadMatrix[i, road.x])
464 {
465 buildingCells[i, road.x] = new BuildingCell(BuildingDirection.None, false);
466 }
467 else
468 {
469 BuildingDirection roadDir = BuildingDirection.None;
470 if (i > road.z)
471 {
472 roadDir = BuildingDirection.Down;
473 }
474 else if (i < road.z)
475 {
476 roadDir = BuildingDirection.Up;
477 }
478
479 buildingCells[i, road.x] = new BuildingCell(roadDir, true);
480 }
481 }
482
483 for (int j = road.x - 1; j <= road.x + 1; j += 2)
484 {
485 if (j < 0 || j >= GameConfig.CHUNK_SIZE || buildingCells[road.z, j] != null)
486 {
487 continue;
488 }
489 if (roadGenerator.RoadMatrix[road.z, j])
490 {
491 buildingCells[road.z, j] = new BuildingCell(BuildingDirection.None, false);
492 }
493 else
494 {
495 BuildingDirection roadDir = BuildingDirection.None;
496
497 if (j > road.x)
498 {
499 roadDir = BuildingDirection.Left;
500 }
501 else if (j < road.x)
502 {
503 roadDir = BuildingDirection.Right;
504 }
505
506 buildingCells[road.z, j] = new BuildingCell(roadDir, true);
507 }
508 }
509 }
510
511 for (int i = 0; i < GameConfig.CHUNK_SIZE; i++)
512 {
513 for (int j = 0; j < GameConfig.CHUNK_SIZE; j++)
514 {
515 if (buildingCells[i, j] is null)
516 {
517 buildingCells[i, j] = new BuildingCell(BuildingDirection.None, true);
518 }
519 }
520 }

534 private void LockAndTryToPlaceBuilding(int index, int row, int col)
535 {
536 if (buildingCells[row, col].GotRoadNext && buildingCells[row, col].Buildable)
537 {
538 Building building = this.buildings[index];
539 List<BuildingCell> neighbourBuildingCells = new List<BuildingCell>();
540 float zPosCorrection = 0f;
541 float xPosCorrection = 0f;
542 if (((int)buildingCells[row, col].RoadDirection) % 2 == 1)
543 {
544 for (int i = building.RowCount / -2; i < Mathf.CeilToInt(building.RowCount / 2f); i++)
545 {
546 if (!buildingCells[row, col + i].Buildable)
547 {
548 return;
549 }
550 neighbourBuildingCells.Add(buildingCells[row, col + i]);
551 }
552 if (building.RowCount % 2 == 0)
553 {
554 xPosCorrection = GameConfig.CHUNK_SIZE / -2;
555 }
556 if (building.ColumnCount % 2 == 0)
557 {
558 zPosCorrection = GameConfig.CHUNK_SIZE / 2;
559 }
560 }
561 else
562 {
563 for (int i = building.RowCount / -2; i < Mathf.CeilToInt(building.RowCount / 2f); i++)
564 {
565 if (!buildingCells[row + i, col].Buildable)
566 {
567 return;
568 }
569 neighbourBuildingCells.Add(buildingCells[row + i, col]);
570 }
571
572 if (building.RowCount % 2 == 0)
573 {
574 zPosCorrection = GameConfig.CHUNK_SIZE / -2;
575 }
576 if (building.ColumnCount % 2 == 0)
577 {
578 xPosCorrection = GameConfig.CHUNK_SIZE / 2;
579 }
580 }
581
582 if (Random.Range(0, 3) == 0)
583 {
584 float rotation = ((int)buildingCells[row, col].RoadDirection) % 4 * 90f;
585 var absolutePosition = GetAbsolutePosition();
586 Vector3 postition = new Vector3(absolutePosition.Item1 + col * GameConfig.CHUNK_SCALE * GameConfig.CHUNK_CELL + xPosCorrection, 0, absolutePosition.Item2 + row * GameConfig.CHUNK_SCALE * GameConfig.CHUNK_CELL + zPosCorrection);
587
588 Addressables.InstantiateAsync(building.AddressableKey, postition, Quaternion.Euler(0, rotation, 0), this.gameObject.transform);
589
590 foreach (var item in neighbourBuildingCells)

592 item.Occupy();
593 }
