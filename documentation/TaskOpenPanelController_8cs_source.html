107 public void OpenUp(PlayerTask taskContainer = null, TaskType type = TaskType.GoodTask)
108 {
109 try

111 //If the task is null create a dummy otherwise load it
112 if (taskContainer is not null)
113 {
114 this.CurrentTask = taskContainer;
115 this.isNewTask = false;
116 //Create a deep copy

119 else
120 {

122 this.CurrentTask.ChangeType(type);
123 this.isNewTask = true;
124 this.playerTaskOnOpen = null;

126
127 taskNameInput.text = this.CurrentTask.TaskName;
128 descriptionInput.text = this.CurrentTask.Description;
129
130 for (int i = 0; i < TASKINTERVALS.Length; i++)
131 {
132 if (TASKINTERVALS[i] == (int)this.CurrentTask.PeriodRate)
133 {
134 taskIntervals.SetValueWithoutNotify(i);
135 break;
136 }
137 }
138 //Show/Hide the delete button
139 deleteTaskButton.gameObject.SetActive(!isNewTask);
140 this.gameObject.SetActive(true);
141 UpdateButtons();
142 }

208 public void Save()
209 {
210 //Collect the values from the inputs
211 CollectFromFields();
212
213 TaskRequest taskRequest = new TaskRequest()
214 {
215 TaskName = this.CurrentTask.TaskName,
216 Description = this.CurrentTask.Description,
217 PeriodRate = (int)CurrentTask.PeriodRate,
218 TaskType = Convert.ToBoolean(CurrentTask.TaskType)

220 //If it is a new one call the create endpoint for the task otherwise the update one
221 if (isNewTask)
222 {
223 CoroutineRunner.RunCoroutine(Server.SendPostRequest<Thesis_backend.Data_Structures.PlayerTask>(ServerConfig.PATHFORTASKCREATE, taskRequest, SavedTask, onFailedAction: uIController.ShowTaskFail));
224 }
225 else
226 {
227 CoroutineRunner.RunCoroutine(Server.SendPatchRequest<Thesis_backend.Data_Structures.PlayerTask>(ServerConfig.PATHFORTASKUPDATE(CurrentTask.ID), taskRequest, SavedTask, onFailedAction: uIController.ShowTaskFail));
228 }
