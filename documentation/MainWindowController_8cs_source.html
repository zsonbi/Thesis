127 public void SortingChanged()
128 {
129 //Create a temporary dictionary which will store the sorted values
130 Dictionary<long, TaskDisplayHandler> newTasks = new Dictionary<long, TaskDisplayHandler>();
131 //Reorder the tasks with the appropriate sorting predicate
132 switch ((TaskSortType)taskSortDropdown.value)
133 {
134 case TaskSortType.Added:
135 foreach (var item in Tasks.OrderBy(x => x.Key))
136 {
137 newTasks.Add(item.Key, item.Value);
138 }
139 break;
140
141 case TaskSortType.Name:
142 foreach (var item in Tasks.OrderBy(x => x.Value.CurrentTask.TaskName))
143 {
144 newTasks.Add(item.Key, item.Value);
145 }
146 break;
147
148 case TaskSortType.Type:
149 foreach (var item in Tasks.OrderBy(x => x.Value.CurrentTask.PeriodRate))
150 {
151 newTasks.Add(item.Key, item.Value);
152 }
153 break;
154
155 case TaskSortType.Available:
156 foreach (var item in Tasks.OrderBy(x => x.Value.CurrentTask.Completed))
157 {
158 newTasks.Add(item.Key, item.Value);
159 }
160 break;
161
162 default:
163 Debug.LogError("No such sorting way is configured");
164 break;
165 }
166 //Override the previous order
167 Tasks = newTasks;
168 //Reload the tasks
169 if (taskOpenPanelController.CurrentTask.TaskType)

171 LoadBadHabits();
172 }

174 {
